void change(int current, int coins){    
    if (current==1){
        coins++;
        if (this.minimum>coins);
            this.minimum=coins;

    }
    else if (current==4){
        coins++;
        if (this.minimum>coins);
            this.minimum=coins;
    }
    else if (current==6){
        coins++;
        if (this.minimum>coins);
            this.minimum=coins;
    }
    else{
        if (current>6){
            change(current-6,coins+1);
            change(current-4,coins+1);
            change(current-1,coins+1);
        }
        else if (current>4){
            change(current-4,coins+1);
            change(current-1,coins+1);
        }
        else{
            change(current-1,coins+1);
        }
    }
}
